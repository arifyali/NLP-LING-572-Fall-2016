\subsection*{code to build confusion matrix and cohen's $\kappa$}
\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlstd{cm} \hlkwb{=} \hlkwd{read.csv}\hlstd{(}\hlstr{"confusion_matrix_POS.csv"}\hlstd{)}
\hlstd{POS_cm} \hlkwb{=} \hlkwa{function}\hlstd{(}\hlkwc{p1}\hlstd{,} \hlkwc{p2}\hlstd{)\{}
  \hlstd{cm_pi} \hlkwb{=} \hlstd{cm[,} \hlkwd{c}\hlstd{(p1,p2)]}
  \hlkwd{levels}\hlstd{(cm_pi[,p1])} \hlkwb{=} \hlkwd{c}\hlstd{(}\hlkwd{levels}\hlstd{(cm_pi[,p1]),}
                         \hlkwd{levels}\hlstd{(cm_pi[,p2])[}\hlopt{-}\hlkwd{which}\hlstd{(}\hlkwd{levels}\hlstd{(cm_pi[,p2])} \hlopt{%in%} \hlkwd{levels}\hlstd{(cm_pi[,p1]))])}
  \hlkwd{levels}\hlstd{(cm_pi[,p2])} \hlkwb{=} \hlkwd{c}\hlstd{(}\hlkwd{levels}\hlstd{(cm_pi[,p2]),}
                         \hlkwd{levels}\hlstd{(cm_pi[,p1])[}\hlopt{-}\hlkwd{which}\hlstd{(}\hlkwd{levels}\hlstd{(cm_pi[,p1])} \hlopt{%in%} \hlkwd{levels}\hlstd{(cm_pi[,p2]))])}

  \hlstd{confusion_matrix} \hlkwb{=} \hlkwd{table}\hlstd{(cm_pi)}
  \hlstd{confusion_matrix} \hlkwb{=} \hlstd{confusion_matrix[}\hlkwd{order}\hlstd{(}\hlkwd{rownames}\hlstd{(confusion_matrix)),}
                                      \hlkwd{order}\hlstd{(}\hlkwd{colnames}\hlstd{(confusion_matrix))]}
  \hlkwd{print}\hlstd{(}\hlstr{"Confusion Matrix"}\hlstd{)}
  \hlkwd{print}\hlstd{(confusion_matrix)}
  \hlstd{p_0} \hlkwb{=} \hlkwd{sum}\hlstd{(}\hlkwd{as.character}\hlstd{(cm_pi[,p1])}\hlopt{==}\hlkwd{as.character}\hlstd{(cm_pi[,p2]))}\hlopt{/}\hlkwd{nrow}\hlstd{(cm_pi)}
  \hlstd{p_e} \hlkwb{=} \hlkwd{sum}\hlstd{(}\hlkwd{rowSums}\hlstd{(confusion_matrix)}\hlopt{/}\hlkwd{nrow}\hlstd{(cm_pi)}\hlopt{*}\hlstd{(}\hlkwd{colSums}\hlstd{(confusion_matrix))}\hlopt{/}\hlkwd{nrow}\hlstd{(cm_pi))}
  \hlcom{#P(A=X) =0 therefore P(A=X)*P(B=X) = 0}

  \hlkwd{print}\hlstd{(}\hlstr{"Agreement rate"}\hlstd{)}
  \hlkwd{print}\hlstd{(p_0)}
  \hlkwd{print}\hlstd{(}\hlstr{"Cohen's Kappa"}\hlstd{)}
  \hlkwd{return}\hlstd{((p_0}\hlopt{-}\hlstd{p_e)}\hlopt{/}\hlstd{(}\hlnum{1}\hlopt{-}\hlstd{p_e))}
\hlstd{\}}
\end{alltt}
\end{kframe}
\end{knitrout}
